                float4 positionOS: POSITION;
                float2 uv: TEXCOORD0;
            };

            {
                float4 positionHCS: SV_POSITION;
                float2 uv: TEXCOORD0;

                float3 toCamera = _WorldSpaceCameraPos - worldPos;
